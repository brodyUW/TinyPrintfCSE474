226 static void formatCommon(char *format, va_list a)
227 {
228  char c;
229  int i;
230  long n;
231  while(c = *format++) {
232  if(c == '%') {
233  switch(c = *format++) {
234  case 's': // String
235  puts(va_arg(a, char*));
236  break;
237  case 'c': // Char
238  putc(va_arg(a, char));
239  break;
240  case 'i': // 16 bit Integer
241  case 'u': // 16 bit Unsigned
242  i = va_arg(a, int);
243  if(c == 'i' && i < 0) i = -i, putc('-');
244  xtoa((unsigned)i, dv + 5);
245  break;
246  case 'l': // 32 bit Long
247  case 'n': // 32 bit uNsigned loNg
248  n = va_arg(a, long);
249  if(c == 'l' && n < 0) n = -n, putc('-');
250  xtoa((unsigned long)n, dv);
251  break;
252  case 'x': // 16 bit heXadecimal
253  i = va_arg(a, int);
254  puth(i >> 12);
255  puth(i >> 8);
256  puth(i >> 4);
257  puth(i);
258  break;
259  case 0: return;
260  default: goto bad_fmt;
261  }
262  } else
263  bad_fmt: putc(c);
264  }
265 }
